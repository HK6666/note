```java
public class InsertionSort{  
  
    public void sort(int[] A){  
        for (int i = 1; i<A.length; i++){  
            // 将 A[i] 插入在卡片0 到卡片i之间  
            // j代表抓到的牌，放到最右侧，不断交换到对应的位置  
  
            int value = A[i];  
            // j是位置  
            int j = i;  
  
            //把所有大于本数的数据往数组后一位移动一位  
            for (; j>0 && A[j-1]>value;j--){  
                A[j] = A[j-1];  
            }  
  
            A[j] = value;  
        }  
  
        for (int number: A){  
            System.out.println("number: "+ number);  
        }  
        System.out.println(A);  
    }  
  
    public static void main(String[] args) {  
        InsertionSort insertionSort = new InsertionSort();  
        int []A = {1,3,4,2,7,6,5,9,8};  
        insertionSort.sort(A);  
  
    }  
}
```

/**  
 * Created by huangkai on 2023/1/10. *///从后往前遍历，把最大的放最后面，然后在剩余数组中找最大的  
public class SelectSort {  
    public void sort(int[] A){  
        for (int i = A.length-1; i>=0; i--){  
            // 0 - A[i] 找到最大值并和最右边交换  
            int j = maxIndex(A,0,i+1);  
            swap(A,i,j);  
        }  
  
        for (int number: A){  
            System.out.println("number: "+ number);  
        }  
        System.out.println(A);  
    }  
  
    //找到当前数组中最大的索引值  
    private int maxIndex(int[] a, int i, int j) {  
        int max = Integer.MIN_VALUE;  
  
        int maxIndex = i;  
        for(int k = 0; k<j;k++){  
            if (max < a[k]){  
                max = a[k];  
                maxIndex = k;  
            }  
        }  
        return maxIndex;  
    }  
  
    private void swap(int[] a, int i, int j) {  
        int temp = a[i];  
        a[i] = a[j];  
        a[j] = temp;  
    }  
  
    public static void main(String[] args) {  
        InsertionSort insertionSort = new InsertionSort();  
        int []A = {1,3,4,2,7,6,5,9,8};  
        insertionSort.sort(A);  
  
    }  
}
